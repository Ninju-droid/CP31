#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun() { 
    ll a,b;
    cin>>a>>b;

    ll x = 0;
    if((a-1)%4==0){
        x = a-1;
    }
    else if((a-1)%4==1){
        x =1;
    }
    else if((a-1)%4==2){
        x=a;
    }
    else{
        x=0;
    }
    if(x==b){
        cout<<a<<endl;
        return;
    }
    else if((x^b)==a){
        cout<<a+2<<endl;
        return;
    }
    else{
        cout<<a+1<<endl;
        return;
    }
}

int main() {
#ifndef ONLINE_JUDGE

    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
