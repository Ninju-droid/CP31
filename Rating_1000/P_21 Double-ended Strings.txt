#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   string a,b;
   cin>>a>>b;
   int n = a.size();
   int m = b.size();
   
   if(n<m){
    swap(a,b);
    swap(n,m);
   }
   ll ans =0;
   ll maxi =0;
   for(ll l =1;l<=m;l++){
    for(ll j=0;j<=n-l;j++){
     for(ll i=0;i<=m-l;i++){
        if(b.substr(i,l)==a.substr(j,l)){
            maxi=max(l,maxi);
        }
     }
    }
   }
   ans = n-maxi+m-maxi;
   cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
