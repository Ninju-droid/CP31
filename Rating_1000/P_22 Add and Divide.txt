#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll a,b;
   cin>>a>>b;
   if(a==0){
    cout<<0<<endl;
    return;
   }
   ll ans = INT_MAX;
   
   for(int i=0;i<=31;i++){
    ll bb = b+i;
    if(bb==1)continue;
    ll cnt = i;
    ll temp = a;
    while(temp>0){
        temp/=bb;
        cnt++;
    }
    ans = min(ans,cnt);
   }

   cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
