#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
  int n,m;
  cin>>n>>m;
  vector<vector<int>>a(n,vector<int>(m));
  int cnt =0;
  int mini = INT_MAX;
  int total = 0;
  for(int i=0;i<n;i++){
    for(int j=0;j<m;j++){
        cin>>a[i][j];
        if(a[i][j]<0){
            cnt++;
            a[i][j]*= -1;
        }
        mini = min(mini,a[i][j]);
        total+=a[i][j];
    }
  }
  if(cnt%2==0){
    cout<<total<<endl;
    return;
  }
  else{
    total-= 2*mini;
    cout<<total<<endl;
    return;
  }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
