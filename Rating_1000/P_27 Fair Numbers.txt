#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
  ll n;
  cin>>n;
  ll dup ;
  while(true){
    dup = n;
    ll l;
    bool isCheck = true;
    while(dup){
        l = dup%10;
        dup/=10;
        if(l==0)continue;
        if(n%l!=0){
            isCheck = false;
            break;
        }
    }
    if(isCheck){
        cout<<n<<endl;
        return;
    }
    n++;
  }

}

int main() {
#ifndef ONLINE_JUDGE

#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
