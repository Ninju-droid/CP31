#include <bits/stdc++.h>
using namespace std;
# define ll long long 

void fun(){
  int n;
  cin>>n;
  int l = sqrt(n);
  int ans = -1;
  int i = 2;
  while(i<=l){
    if(n%i==0){
      ans = i;
      break;
    }
    i++;
  }
  if(ans == -1){
    cout<<1<<" "<<n-1<<endl;
  }
  else{
    ans = n/ans;
    cout<<ans<<" "<<n-ans<<endl;
  }

}
int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
