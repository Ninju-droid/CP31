#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    ll n;
    cin>>n;
    vector<ll>a(n),aa(n);
    for(ll i=0;i<n;i++)cin>>a[i];
    for(ll i=0;i<n;i++)cin>>aa[i];
    ll l=0,r=n-1;    
    while(a[l]==aa[l] ||a[r]==aa[r]){
        if(a[l]==aa[l])l++;
        if(a[r]==aa[r])r--;
    }    
    while(aa[r]<=aa[r+1] && r<n-1)r++;
    while(aa[l]>=aa[l-1] && l>0)l--;

    l++;r++;
    cout<<l<<" "<<r<<endl;   
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
