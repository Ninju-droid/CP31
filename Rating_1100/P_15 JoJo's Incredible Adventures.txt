#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
     string s;
     cin>>s;
     ll n = s.size();
     ll start=0,last=0,maxi=0;
     ll cnt =0;
     bool isAllOne = true;
     for(int i=0;i<n;i++){
        if(s[i]=='1'){
            cnt++;
            maxi = max(maxi,cnt);
        }
        else{
            cnt=0;
            isAllOne = false;
        }
     }
     if(isAllOne){
        cout<<n*n<<endl;
        return;
     }
     ll i=0;
     while(s[i]=='1' && i<n){
        start++;
        i++;
     }
     i = n-1;
     while(s[i]=='1' && i>=0){
        last++;
        i--;
     }
     maxi++;
     ll ans = (maxi/2)*(maxi - maxi/2);
     maxi = start+last+1;
     ans = max((maxi/2)*(maxi - maxi/2),ans);
     cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
