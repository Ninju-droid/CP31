#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n,k;
   cin>>n>>k;
   vector<vector<ll>>a(n,vector<ll>(n));
   for(ll i=0;i<n;i++){
    for(ll j=0;j<n;j++){
        cin>>a[i][j];
    }
   }
   ll cnt = 0;
for (ll i = 0; i < n; ++i) {
    for (ll j = 0; j < n; ++j) {
        ll ii = n - 1 - i;
        ll jj = n - 1 - j;
        if (i > ii || (i == ii && j >= jj)) continue; // skip duplicates or center
        if (a[i][j] != a[ii][jj]) cnt++;
    }
}
   if(k>=cnt && (k%2==cnt%2 || n%2==1)){
    cout<<"yes"<<endl;
   } 
   else{
    cout<<"no"<<endl;
   } 
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
