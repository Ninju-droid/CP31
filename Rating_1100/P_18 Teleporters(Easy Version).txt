#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n,c;
   cin>>n>>c;
   vector<ll>a(n);
   for(ll i=0;i<n;i++){
    cin>>a[i];
    a[i]+=(i+1);
   }
   sort(a.begin(),a.end());
   ll total = 0;
   for(ll i=0;i<n;i++){
    total+=a[i];
    if(total>c){
        cout<<i<<endl;
        return;
    }
   }
    cout<<n<<endl; 
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
