#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   cin>>n;
   vector<ll>a(n);
   ll cnt=0;
   ll total = 0;
   ll mini = INT_MAX;
   for(ll i=0;i<n;i++){
    cin>>a[i];
    if(a[i]<0){
        cnt++;
        a[i] = -a[i];
    }
    mini = min(mini,a[i]);
    total+=a[i];
   }
    if(cnt%2==0){
        cout<<total<<endl;
    } 
    else{
        cout<<total-2*mini<<endl;
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
