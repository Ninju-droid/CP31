#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   cin>>n;
   vector<ll>a(n);
   for(ll i=0;i<n;i++)cin>>a[i];

   vector<ll>pre(n),suff(n);
   pre[0] = a[0];
   for(int i=1;i<n;i++){
    pre[i]=pre[i-1]+a[i];
   } 
   suff[n-1]=a[n-1];
   for(ll i=n-2;i>=0;i--){
    suff[i]=a[i]+suff[i+1];
   }
   ll ans = INT_MIN;
   for(ll i=1;i<n;i++){
        ll g = gcd(pre[i-1],suff[i]);
        ans = max(ans,g);
   }
    cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
