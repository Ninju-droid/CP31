#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    ll n,l,r;
    cin>>n>>l>>r;
    vector<ll>a(n);
    for(ll i=1;i<=n;i++){
       a[i-1] = ((l-1)/i +1)*i;
       if(a[i-1]<l || a[i-1]>r){
        cout<<"NO"<<endl;
        return;
       }
    }
    cout<<"YES"<<endl;
    for(ll i=0;i<n;i++){
        cout<<a[i]<<" ";
    }
    cout<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
