#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    ll n;
    cin>>n;
    ll ans = (1<<30)-1;
    for(int i=0;i<n;i++){
        ll x;
        cin>>x;
        if(x!=i){
            ans&=x;
        }
    }
    cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
