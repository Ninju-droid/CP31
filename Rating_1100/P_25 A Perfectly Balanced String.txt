#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    string s;
    cin>>s;
    set<char>st(s.begin(),s.end());
    ll n = s.size();
    ll k = st.size();
    for(ll i= k;i<n;i++){
        if(s[i]!=s[i-k]){
            cout<<"NO"<<endl;
            return;
        }
    }
    cout<<"YES"<<endl;
    return;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
