#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   cin>>n;
   vector<ll>a(n);
   for(ll i=0;i<n;i++)cin>>a[i];
   ll ans = 0; 
   ll l=0,r=n-1;
   ll al =a[l],b=a[r];
   while(l<r){
      if(al>b){
        r--;
        b+=a[r];
      }
      else if(b>al){
        l++;
        al+=a[l];
      }
      else{
        ans = (l+(n-r)+1);
        l++;
        r--;
        if(l<r){
            al+=a[l];
            b+=a[r];}
        }
   } 
 cout<<ans<<endl;  
} 

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
