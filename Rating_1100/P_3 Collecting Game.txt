#include <bits/stdc++.h>
using namespace std;
# define ll long long 

void fun(){
   ll n;
   cin>>n;
   vector<pair<ll,ll>>a(n);
   vector<ll>ans(n);
   vector<ll>presum(n);
   for(ll i=0;i<n;i++){
      ll x;
      cin>>x;
      a[i] = {x,i};
   }
   sort(a.begin(),a.end());
   presum[0] = a[0].first;
   for(int i=1;i<n;i++){
    presum[i] = presum[i-1]+a[i].first;
   }

   ll j = 0;
   ll index = 0;
   for(ll i=0;i<n-1;i++){
    if(j<i) j = i;
    index = a[i].second;
    while(presum[j]>=a[j+1].first && j<n-1)j++;
    ans[index]=j;
   }
   index = a[n-1].second;
   ans[index] = n-1;
   for(ll i =0;i<n;i++)cout<<ans[i]<<" ";

    cout<<endl;
}
int main() {
#ifndef ONLINE_JUDGE

    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
