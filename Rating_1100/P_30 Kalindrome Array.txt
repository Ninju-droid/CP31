#include <bits/stdc++.h>
using namespace std;
#define ll long long 

bool check(vector<ll>&a,ll x){
  vector<ll>b;
  ll n = a.size();
  for(ll i=0;i<n;i++){
    if(a[i]!=x)b.push_back(a[i]);
  }
  ll m = b.size();
  for(ll i =0;i<m;i++){
    if(b[i]!=b[m-1-i]){
      return false;
    }
  }
  return true;
}

void fun(){ 
   ll n;
   cin>>n;
   vector<ll>a(n);
   for(ll i=0;i<n;i++)cin>>a[i];

   for(ll i=0;i<n;i++){
    if(a[i] != a[n-1-i]){
        if(check(a,a[i]) || check(a,a[n-i-1])){
                cout<<"YES"<<endl;
        }
    else{
        cout<<"NO"<<endl;
    }
    return;
   } 
  }
  cout<<"YES"<<endl;
} 

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
