    ll n,q;
    cin>>n>>q;
    vector<ll>c(51,0);
    vector<ll>a(n);
    for(ll i=0;i<n;i++){
        cin>>a[i];
        if(c[a[i]]==0){
            c[a[i]]=i+1;
        }
    }
    for(ll i=0;i<q;i++){
        ll test ;cin>>test;
        cout<<c[test]<<" ";
        ll index = c[test];
        for(ll i =1;i<=50;i++){
            if(c[i]<index)c[i]++;
        }
        c[test] = 1;
    }
cout<<endl;

}
