#include <bits/stdc++.h>
using namespace std;
#define ll long long

void fun() {
   ll n,q;
   cin>>n>>q;
   vector<ll>a(n),b(q);
   for(int i=0;i<n;i++)cin>>a[i];
   for(int i=0;i<q;i++)cin>>b[i]; 
  
   ll prev = 31;
   for(int i=0;i<q;i++){
      if(b[i]>=prev) continue;
      int div = pow(2,b[i]);
      for(int j=0;j<n;j++){
        if(a[j]%div==0){
            a[j]+=div/2;
        }
      }
   } 
   
  for(int i=0;i<n;i++){
    cout<<a[i]<<" ";
  } 
cout<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) {
        fun();
    }
    return 0;
}
