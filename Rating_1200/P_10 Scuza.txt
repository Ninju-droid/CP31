#include <bits/stdc++.h>
using namespace std;
# define ll long long 

void fun(){
   int n,q;
   cin>>n>>q;
   vector<ll>a(n);
   vector<ll>query(q);
   vector<ll>ans;
   for(int i=0;i<n;i++)cin>>a[i];
   for(int i=0;i<q;i++)cin>>query[i];
   vector<ll>max_step(n),sum_step(n);
	max_step[0] = a[0];
	sum_step[0] = a[0];
   for(int i=1;i<n;i++){
   	max_step[i]=max(max_step[i-1],a[i]);
   	sum_step[i]=sum_step[i-1]+a[i];
   }
   for(int i=0;i<q;i++){
   	ll k = query[i];
   	if(max_step[n-1]<=k){
   		cout<<sum_step[n-1]<<" ";
   		continue;
   	}
   	if(max_step[0]>k){
   		cout<<0<<" ";
   		continue;
   	}
   	ll low = 0 ,high = n-1;
   	while(low<high){
   		ll mid = low +(high-low)/2;
   		if(max_step[mid]<=k)low = mid+1;
   		else high = mid; 
   	}
   	cout<<sum_step[low-1]<<" ";
   }
   cout<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
