#include <bits/stdc++.h>
using namespace std;
# define ll long long 
# define mod 1000000007
void fun() {
    ll n,m;
    cin >>n>>m;
    vector<ll>a(m);
    for(ll i =0;i<m;i++)cin>>a[i];
    sort(a.begin(),a.end());    
    vector<ll>gaps(m);
    for(ll i =0;i<m;i++){
        if(i==0){
            gaps[i]=n-a[m-1]+a[0]-1;
        }
        else gaps[i] = a[i]-a[i-1]-1;
    }
    sort(gaps.begin(),gaps.end(),greater<>());
    ll time = 0;
    ll saved = 0;
    for(ll i =0;i<m;i++){
        ll gap = gaps[i];
        ll effect = 2*time;
        if(gap-effect==1||gap-effect==2){
            saved+=1;
            time++;
        }
        else if(gap-effect > 2){
            saved+=(gap-effect-1);
            time+=2;
        }
    }   
   cout<<n-saved<<endl; 
}
int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
