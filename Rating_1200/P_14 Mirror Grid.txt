#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   cin>>n;
   vector<string>a(n);
   for(ll i=0;i<n;i++)cin>>a[i];

   ll ans = 0;
   for(ll i=0;i<n/2;i++){
    for(ll j=i;j<n-i-1;j++){
        ll one=0,zero=0;
        if(a[i][j]=='1')one++;
        else zero++;
        if(a[j][n-1-i]=='1')one++;
        else zero++;
        if(a[n-1-i][n-1-j]=='1')one++;
        else zero++;
        if(a[n-1-j][i]=='1')one++;
        else zero++;

        ans += min(one,zero);
    }
   } 
  cout<<ans<<endl;
} 

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
