#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n,s;
   cin>>n>>s;
   vector<ll>a(n);
   ll sum = 0;
   for(ll i=0;i<n;i++){
    cin>>a[i];
    sum+=a[i];
   }
   if(sum==s){
    cout<<0<<endl;
    return;
   }
   if(sum<s){
    cout<<-1<<endl;
    return;
   } 
   ll ans = 0; 
   sum = 0; 
   ll index=0;
   ll j=0;
   while(index<n){
    sum+=a[index];
    if(sum==s){
        ans = max(ans,index-j+1);
    }
    if(sum>s){
        while(sum>s){
            sum-=a[j];
            j++;
        }
    }
    index++;
   } 
   cout<<n-ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
