#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   cin>>n;
   vector<ll>a(n);
   for(ll i=0;i<n;i++)cin>>a[i];
   if(a[n-2]>a[n-1]){
    cout<<-1<<endl;
    return;
   } 
   ll maxi = max(a[n-1],a[n-2]);
   ll ind =(maxi==a[n-1])?n-1: n-2;
   vector<vector<ll>>ans;
   for(ll i=n-3;i>=0;i--){
     if(a[i]>a[i+1]){
        a[i] = a[i+1]-maxi;
        ans.push_back({i+1,i+2,ind+1});
        if(a[i]>a[i+1]){
            cout<<-1<<endl;
            return;
        }
     }
   }
   ll m = ans.size();
   cout<<m<<endl;
   for(ll i=0;i<m;i++){
    cout<<ans[i][0]<<" "<<ans[i][1]<<" "<<ans[i][2]<<endl;
   }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
