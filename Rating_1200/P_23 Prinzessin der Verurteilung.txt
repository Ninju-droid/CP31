    bool ok = false;
     for(ll i=0;i<n;i++){

            break;
        }
     }
     if(!ok){
        cout<<ch<<endl;
        return;
     }
   }
   queue<string>q;
   for(char ch = 'a';ch<='z'; ch++){
     for(char c = 'a';c<='z';c++){
        string two = "";
        two+=ch;
        two+=c;
        if(a.find(two) != std::string::npos){
            q.push(two);
        }

        }
     }
   }

   while(!q.empty()){

    for(char ch = 'a';ch<='z';ch++){

        two+=ch;

   }

}
