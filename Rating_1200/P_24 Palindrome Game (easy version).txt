#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   string s;
   cin>>n;
   cin>>s;
   ll num = 0;
   for(ll i=0;i<n;i++){
    if(s[i]=='0')num++;
   }
   if(num%2 == 0){
     cout<<"BOB"<<endl;
   }
   else{
    if(num==1)cout<<"BOB"<<endl;
    else cout<<"ALICE"<<endl;
   }
   
}

int main() {

}
