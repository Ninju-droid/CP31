#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
  ll n;
  cin>>n;
  vector<ll>a(n);
  map<ll,ll>mp;
  for(ll i=0;i<n;i++){
    ll x;
    cin>>x;
    a[i] = x-i;
    mp[a[i]]++;
  }
  ll ans =0;
  for(auto it :mp){
    ll cnt = it.second;
    ans += (cnt*(cnt-1)/2);
  }
    cout<<ans<<endl;  
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
