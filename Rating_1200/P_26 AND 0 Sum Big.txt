#include <bits/stdc++.h>
using namespace std;
#define ll long long 
const ll mod = 1e9+7 ;
void fun(){ 
  ll n,k;
  cin>>n>>k;
  ll ans=1;
  while(k--){
    ans = ((ans%mod)*(n%mod))%mod;
  }
  cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
