#define ll long long 

void fun(){ 
  ll n;
  cin>>n;
  string a,b;
  cin>>a>>b;
  ll a_z=0,a_o=0,b_z=0,b_o=0;
  for(ll i=0;i<n;i++){
    if(a[i]=='0')a_z++;
    else a_o++;

    if(b[i]=='0')b_z++;
    else b_o++;

    if(a_z!=b_z){
        if(a_z!=b_o){
            cout<<"NO"<<endl;
            return;
        }
    }
  }
  if(a_z==b_z) cout<<"YES"<<endl;
  else cout<<"NO"<<endl;
}
