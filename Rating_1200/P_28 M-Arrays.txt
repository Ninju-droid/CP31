#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
  ll n,m;
  cin>>n>>m;
  map<ll,ll>mp;
  ll ans = 0;
  for(ll i=0;i<n;i++){
    ll x;
    cin>>x;
    mp[x%m]++; 
  }
  for(auto &it :mp){
    if(it.first==0 || 2*it.first==m){
      ans++;
    }
    else if(2*it.first<m || mp.find(m-it.first)==mp.end()){
      ll cnt = it.second;
      ll cntM = mp[m-it.first];
      ans+= 1 + max(0ll,abs(cnt-cntM)-1);
    }
  }
 cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
