#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n;
   cin>>n;
   vector<ll>a(n-1);
   for(ll i=0;i<n-1;i++)cin>>a[i];
   string s;
   cin>>s;
   ll cnt =0;
   vector<pair<ll,ll>>ans(n);
   for(ll i=0;i<n;i++){
     if(s[i]=='W')ans[i]={1,0};
     else ans[i] = {0,1};
   }

   for(ll i=n-1;i>=1;i--){
     ll wh = ans[i].first;
     ll bl = ans[i].second;
     if(wh==bl) cnt++;

     ll root = a[i-1]-1;
     ans[root].first+=wh;
     ans[root].second+=bl;
   }
   if(ans[0].first==ans[0].second)cnt++;
   cout<<cnt<<endl;

}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
