#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
   ll n,k;
   cin>>n>>k;
   vector<ll>a(n);
   for(ll i=0;i<n;i++)cin>>a[i];
   vector<ll>mp(31);
   for(ll i=0;i<n;i++){
     ll val = a[i];
     for(ll j=0;j<=30;j++){
        if(val & (1<<j)) mp[j]++;
     }
   }
   ll ans =0;
   for(ll i=30;i>=0;i--){
    ll cnt = mp[i];
    if(cnt==n)ans+=pow(2,i);
    else if(n-cnt <=k){
        ans+=pow(2,i);
        k-=(n-cnt);
    }
   }
   cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
