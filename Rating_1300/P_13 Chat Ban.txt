#include <bits/stdc++.h>
using namespace std;
#define ll long long 

ll calculateSumOfFirstX(ll x){
    return x*1ll*(x+1)/2;
}

void fun(){ 
    ll k,x; cin>>k>>x;
    ll low = 1, high = 2*k-1,ans = 2*k-1;
    while(low<=high){
        ll mid = low+(high-low)/2;
        ll cnt;
        if(mid>=k){
            cnt = calculateSumOfFirstX(k)+calculateSumOfFirstX(k-1)-calculateSumOfFirstX(2*k-1-mid);  
        }
        else{
            cnt = calculateSumOfFirstX(mid);
        }
        if(cnt>=x){
            ans = mid;
            high = mid-1;
        }
        else{
            low = mid+1;
        }
    }
cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
