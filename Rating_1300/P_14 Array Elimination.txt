#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    ll n;
    cin>>n;
    vector<ll>a(n);
    for(ll i=0;i<n;i++)cin>>a[i];
    vector<ll>ans(n+1,1);
    map<ll,ll>mp;
    for(ll i=0;i<n;i++){
        for(ll j=0;j<=30;j++){
            if(a[i]&(1<<j))mp[j]++;
        }
    }
    for(auto &it:mp){
        ll val = it.second;
        for(ll i=1;i<=n;i++){
            if(val%i!=0)ans[i]=0;
        }
    }
    for(ll i=1;i<=n;i++){
        if(ans[i])cout<<i<<" ";
    }
    cout<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
