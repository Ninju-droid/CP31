#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    ll n;
    cin>>n;
    vector<vector<ll>>a(n);
    vector<pair<ll,ll>>mp(n);
    for(ll i=0;i<n;i++){
    ll k; cin >> k;
        vector<ll> aa(k);
        for(ll j=0;j<k;j++) cin >> aa[j];

        ll need = 0;
        for(ll j=0;j<k;j++){
            need = max(need, aa[j] - j + 1); 
        }

        a[i] = aa;
        mp[i]={need, i};       
    }
    sort(mp.begin(),mp.end());
    ll ans = 0;
    ll cnt = 0;
    for(ll i=0;i<n;i++){
        ll need = mp[i].first;
        ll index = mp[i].second;
        if(cnt<need){
            ans+=(need-cnt);
            cnt = need;
        }
        cnt+=a[index].size();
    }
    cout<<ans<<endl;    
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
