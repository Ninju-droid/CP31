#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
    ll n,w;
    cin>>n>>w;
    map<ll,ll>mp;
    vector<ll>a(n);
    for(ll i=0;i<n;i++) cin>>a[i];
    sort(a.rbegin(),a.rend());    
    vector<ll>widthCount(21,0);
    for(ll i=0;i<n;i++){
        ll logval = log2(a[i]);
        widthCount[logval]++;
    }
    ll height = 0;
    ll countOfUsedBoxes = 0;
    while(countOfUsedBoxes<n){
        ll temp = w;
        for(ll i=20;i>=0;i--){
            while(widthCount[i]>0 && (1<<i) <=temp){
                temp-= (1<<i);
                widthCount[i]--;
                countOfUsedBoxes++;
            }
        }
        height++;
    }
  cout<<height<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
