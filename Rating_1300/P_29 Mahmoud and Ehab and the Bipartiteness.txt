#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define endl '\n'

void fun(){
   ll n;
    cin >> n;
    vector<vector<ll>> g(n+1);
    for (ll i = 0; i < n-1; i++) {
        ll u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vector<int> c(n+1, -1); 
    queue<int> q;
    c[1] = 0;
    q.push(1);

    ll cntA = 0, cntB = 0;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        if (c[u] == 0) cntA++;
        else cntB++;

        for (auto v : g[u]) {
            if (c[v] == -1) {
                c[v] = 1 - c[u];
                q.push(v);
            }
        }
    }

    ll ans = cntA * cntB - (n - 1);
    cout << ans << endl;
}
 
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t=1;
    //cin >> t;
    while (t--) fun();
    return 0;
}
