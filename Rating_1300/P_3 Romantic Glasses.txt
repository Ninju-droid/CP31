void fun() {
    ll n;
    cin >>n;
    vector<ll>a(n);
    for(ll i =0;i<n;i++)cin>>a[i];
    for(ll i= 1;i<n;i+=2){
        a[i] = -a[i];
    }  
    map<ll,ll>m;
    m[0] = 1;
    ll s = 0;
    for(ll i =0;i<n;i++){
        s+=a[i];
        if(m[s]){
            cout<<"yes"<<endl;
            return;
        }
        m[s]++;
    }  

    cout<<"no"<<endl;
}
