#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define mod 1000000007

void fun() {
    ll n;
    cin>>n;
    vector<ll>a(n);
    for(ll i =0;i<n;i++)cin>>a[i];
    map<ll,ll>m;
    for(ll i = 0;i<n;i++){
        ll num = a[i];

        while(num%2==0){
            m[2]++;
            num/=2;
        }
        for(int j=3;j*j<=num;j+=2){
            while(num%j==0){
                m[j]++;
                num/=j;
            }
        }
        if(num>1){
            m[num]++;
        }
    }
    for(auto it:m){
        if(it.second % n != 0){
            cout<<"no"<<endl;
            return;
        }
    }
    cout<<"yes"<<endl;   
}
int main() {
#ifndef ONLINE_JUDGE

#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
