#include <bits/stdc++.h>
using namespace std;
#define ll long long 
ll Mod = 998244353;
void fun(){ 
  string s; cin>>s;
  ll n = s.size();
  ll ans =1;
  ll ansLen =1;
  ll cur =1;
  for(ll i=1;i<n;i++){
    if(s[i] != s[i-1]){
      ansLen++;
      ans = (ans*cur)%Mod;
      cur = 1;
    } else{
      cur++;
    }
  }
  ans = (ans*cur)%Mod;
  for(ll i=1;i<= n-ansLen;i++){
    ans = (ans*i)%Mod;
  }
  cout<<n-ansLen<<" "<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
