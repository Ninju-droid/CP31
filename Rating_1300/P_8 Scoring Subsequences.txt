#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
  ll n;
  cin>>n;
  vector<ll>a(n);
  for(ll i=0;i<n;i++)cin>>a[i];

  vector<ll>res;
  for(ll i=0;i<n;i++){
    ll l=1,r=i+1;
    while(l<=r){
        ll m = (l+r)/2;
        if(a[i-m+1]>=m){
            l = m+1;
        }
        else{
            r = m-1;
        }
    }
    res.push_back(r);
  } 
  for(auto i:res){
    cout<<i<<" ";
  } 
  cout<<endl;
    
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
