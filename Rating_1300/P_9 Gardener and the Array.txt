#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){ 
  ll n;
  cin>>n;
  vector<vector<ll>>a(n);
  map<ll,ll>mp;
  for(ll i=0;i<n;i++){
    ll k;
    cin>>k;
    for(ll j=0;j<k;j++){
        ll val;
        cin>>val;
        a[i].push_back(val);
        mp[val]++;
    }
  }
 bool is_all = true;
 for(ll i=0;i<n;i++){
    for(auto it:a[i]){
        if(mp[it]==1) is_all = false;
    }
    if(is_all){
        cout<<"YES"<<endl;
        return;
    }
    is_all = true;
 }
   cout<<"NO"<<endl; 
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
