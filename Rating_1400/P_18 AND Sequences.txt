#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define endl '\n'
ll mod = 1e9+7 ;

long long fact(ll n) {
    long long result = 1;
    for (long long i = 1; i <= n; i++) {
        result *= i;
        result%=mod;
    }
    return result;
}

void fun(){
    ll n; cin>>n; 
    vector<ll>a(n);
    for(ll i=0;i<n;i++)cin>>a[i]; 
    ll val = a[0];
    for(ll i=0;i<n;i++){
        val = val&a[i];
    } 
    ll cnt = 0;  
    for(ll i=0;i<n;i++){
        if(val==a[i])cnt++;
    } 
    if(cnt<2){
        cout<<0<<endl;
        return;
    }

    ll ans = (((cnt*(cnt-1))%mod)*(fact(n-2)%mod))%mod;
    cout<<ans<<endl;    
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
