#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define endl '\n'

void fun() {
    ll n;cin>>n; 
    vector<ll>a(n);
    for(ll i=0;i<n;i++)cin>>a[i]; 
    string s;cin>>s; 
    vector<ll>pre(n+1,0);
    ll x=0,one=0;
    for(ll i=1;i<=n;i++){
        x ^= a[i-1];
    }
    for(ll i=1;i<=n;i++){
        if(s[i-1]=='1')one^=a[i-1];
        
        pre[i] = pre[i-1]^a[i-1];
    }
    ll q;cin>>q;
    while(q--){
        ll o;cin>>o;
        if(o==1){
            ll a,b;cin>>a>>b; 
            one^= (pre[b]^pre[a-1]);
        }
        else{
            ll a;cin>>a; 
            if(a==0)cout<<(x^one)<<" "; 
            else cout<<one<<" ";
        }
    }
    cout<<endl;
}

int main() {
    ios_base::sync_with_stdio(false);

}
