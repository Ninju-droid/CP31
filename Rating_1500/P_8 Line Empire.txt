#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll mod = 1e9+7;

void fun() {
    ll n,a,b;cin>>n>>a>>b; 
    vector<ll>vec(n+1,0);
    ll ans = 0 , cur =0;
    for(ll i=1;i<=n;i++) cin>>vec[i]; 

    for(ll i=1;i<=n;i++){
        ans += b*(vec[i]-vec[cur]);
        if(a<(n-i)*b && cur+1<n){
            ans+= a*(vec[cur+1]-vec[cur]);
            cur++;
        }
    }
    cout<<ans<<endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
#ifndef ONLINE_JUDGE

    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
}
