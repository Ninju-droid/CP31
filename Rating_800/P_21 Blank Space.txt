#include <bits/stdc++.h>

# define ll long long 

void fun(){
    ll n;
    cin>>n;
    vector<ll>a(n);
    for(ll i=0;i<n;i++)cin>>a[i];
    ll maxi = 0;
    ll cnt =0;
    for(ll i=0;i<n;i++){
        if(a[i]==0){
            cnt++;
        }
        else cnt =0;
        maxi = max(cnt,maxi);
    } 
cout<<maxi<<endl;
}
