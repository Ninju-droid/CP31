#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
    int n;
    cin>>n;
    string s;
    cin>>s;
    int l =0;
    int r = n-1;
    while(l<=r && s[l]!=s[r]){
        l++;
        r--;
        if(l==r){
            cout<<1<<endl;
            return;
        }
    }
    cout<<r-l+1<<endl;
}
