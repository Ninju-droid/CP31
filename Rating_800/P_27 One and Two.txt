void fun(){

   cin>>n;
   vector<int>a(n);
   int total = 0;
   for(int i=0;i<n;i++){
     cin>>a[i];
     if(a[i]==2)total++;
   }
   if(total%2!=0){
    cout<<-1<<endl;
    return;
   }
   int i;
   int k = total/2;
   for(i=0;i<n;i++){
        if(a[i]==2)k--;
        if(k==0)break;
   }
cout<<i+1<<endl;
}
