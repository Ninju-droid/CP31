void fun(){
   int n;
   cin>>n;
   vector<int>a(n);
   for(int i=0;i<n;i++)cin>>a[i];
   sort(a.rbegin(),a.rend());
   if(a[0]==a[1]){
     if(n==2){
        cout<<"no"<<endl;
        return;
     }
     int check = 0;
     for(int i = 2;i<n;i++){
        if(a[i]!=a[0]){
            int temp = a[0];
            a[0] = a[i];
            a[i] = temp;
            check = 1;
        }
     }
     if(!check){
        cout<<"no"<<endl;
        return;
     }
   }

    cout<<"yes"<<endl;
    for(int i=0;i<n;i++)cout<<a[i]<<" ";
    
    cout<<endl;
   
}
