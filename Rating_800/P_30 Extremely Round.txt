#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
   ll n;
   cin>>n;
   int cnt =0;
   while(n>=10){
     cnt++;
     n/=10;
   }
   cout<<cnt*9+n<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
