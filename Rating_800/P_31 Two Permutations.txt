using namespace std;
#define ll long long 

void fun(){
   ll n,a,b;
   cin>>n>>a>>b;
   if(n==a && a==b){
    cout<<"yes"<<endl;
    return;
   }
   if(n-a > b+1){
    cout<<"yes"<<endl;
    return;
   }
   else{
    cout<<"no"<<endl;
    return;
   }
}
