#include <bits/stdc++.h>
using namespace std;
# define ll long long 

void fun(){
  ll n;
  cin>>n;
  vector<ll>a(n);
  ll ans = 0;
  for(ll i=0;i<n;i++)cin>>a[i];

  for(ll i= n-2;i>=0;i--){
    while(a[i]>=a[i+1] && a[i]>0){
        a[i]/=2;
        ans++;
    }
    if(a[i]==a[i+1]){
        cout<<-1<<endl;
        return;
    }
  } 
  cout<<ans<<endl; 
}
int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
