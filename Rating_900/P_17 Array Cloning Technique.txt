#include <bits/stdc++.h>
using namespace std;
# define ll long long 

void fun(){
  ll n;
  cin>>n;
  vector<ll>a(n);
  ll ans = 0;
  map<ll,ll>mp;
  for(ll i =0;i<n;i++){
    cin>>a[i];
    mp[a[i]]++;
  }
ll freq = LLONG_MIN;

for(auto it:mp){
  ll maxi = it.second;
  freq = max(maxi,freq);
}

  ans +=(n-freq);
  while(freq<n){
    ans++;
    freq*=2;
  } 
  cout<<ans<<endl; 
}
int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
