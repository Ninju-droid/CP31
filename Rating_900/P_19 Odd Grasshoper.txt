#define ll long long 

void fun(){
  ll a,n;
  cin>>a>>n;
   ll s =0;
    if(n%4==0){
        s=0;
    }
    else if(n%4==1){
        s = n;
    }
    else if(n%4==2){
        s = 1;
    }
    else{
       s = n+1; 
    }
  
  if((a+(n+2)/2)%2==0){
    cout<<a+s<<endl;
  }
  else{
    cout<<a-s<<endl;
  }
}
