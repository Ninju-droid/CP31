#include<bits/stdc++.h>
using namespace std;

void fun(){
 int n,k;
 cin>>n>>k;
 string s;
 cin>>s;
 vector<int>mp(26,0);
 for(int i=0;i<n;i++){
    mp[s[i]-'a']++;
 }
 int cnt =0;
 for(int i=0;i<26;i++){
   if(mp[i]%2)cnt++;
 }
 if(cnt-1 <= k){
   cout<<"YES"<<endl;
   return;
 }
 if(cnt>k){
   if(cnt-k > 1){
      cout<<"NO"<<endl;
   }
   else cout<<"YES"<<endl;
   return;
 }

}

int main() {

   }

}
