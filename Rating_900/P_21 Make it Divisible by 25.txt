const int MOD = 1e9+7;
const int MAXN = 1e6 + 5;
#define ll long long
#define all(x) (x).begin(), (x).end()
#define allr(x) (x).rbegin(), (x).rend()

ll modpow(ll base, ll exp) {
    ll result = 1;
    base %= MOD;
    while (exp > 0) {
        if (exp & 1) result = (result * base) % MOD;
        base = (base * base) % MOD;
        exp >>= 1;
    }
    return result;
}
ll modinv(ll a){
    return modpow(a,MOD-2);
}
vector<ll> fact(MAXN);
void precomputeFact(){
    fact[0] = 1;
    for(int i=1; i<MAXN; i++){
        fact[i] = (fact[i-1]*i) % MOD;
    }
}
ll nCr(int n,int r){
    if(r>n || r<0) return 0;
    ll denom = (fact[r] * fact[n-r]) % MOD;
    return (fact[n]*modinv(denom)) % MOD;
}
    
bool func(ll m,vector<ll>&a,vector<ll>&b,ll n){ 
    ll cnt = 0 ; 
    for(ll i =1;i<=n;i++){ 
        if(b[i]>=cnt && a[i]>= m-cnt-1)cnt++;
    } 
    return cnt>=m;
}
