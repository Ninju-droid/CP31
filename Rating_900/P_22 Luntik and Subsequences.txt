#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll n;
  cin>>n;
  vector<ll>a(n);
  for(ll i=0;i<n;i++)cin>>a[i];
  ll ans =0;
  for(ll i=0;i<n;i++){
    if(a[i]==1)ans++;
  }  
  ll cnt =0;
  for(ll i=0;i<n;i++){
    if(a[i]==0)cnt++;
  }
  cnt = pow(2,cnt);
cout<<ans*cnt<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
