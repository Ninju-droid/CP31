#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll n;
  cin>>n;
  vector<ll>a(n);
  ll ans = 0;
  for(ll i=0;i<n;i++){
    cin>>a[i];
    ans = max(ans,a[i]);
  }
  for(ll i=0;i<n-1;i++){
    
    for(ll j=i+1;j<n;j++){
        ll low = i,high = j;
        while(low<=high){
            ll num = a[low]&a[high];
            a[low]=num;
            a[high]=num;
            low++;
            high--;
        }
        ll maxi =0;
        for(int i=0;i<n;i++){
            maxi=max(maxi,a[i]);
        }
        ans = min(ans,maxi);
    }
  }
  cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
