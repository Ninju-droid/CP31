#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll a,b;
  cin>>a>>b;
  if(a==b){
    cout<<0<<" "<<0<<endl;
    return;
  }
  if(a>b)swap(a,b);
  ll ans = b-a;
  ll next = ((a+ans)/ans)*ans;
  ll prev = (a/ans)*ans;
  ll diff = min(next-a,a-prev);
  cout<<ans<<" "<<diff<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
