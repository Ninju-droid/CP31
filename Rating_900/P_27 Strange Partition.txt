#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll n,x;
  cin>>n>>x;
  vector<ll>a(n);
  ll minTotal = 0;
  ll maxTotal =0;
  for(ll i=0;i<n;i++){
    cin>>a[i];
    minTotal+=a[i];
    maxTotal+=((a[i]+x-1)/x);
  }
  minTotal+=(x-1);
  minTotal/=x;
  cout<<minTotal<<" "<<maxTotal<<endl;


}

int main() {
#ifndef ONLINE_JUDGE

    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
