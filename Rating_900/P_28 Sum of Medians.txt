#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll n,k;
  cin>>n>>k;
  vector<ll>a(n*k);
  for(ll i=0;i<n*k;i++){
    cin>>a[i];
  }
  ll cnt = (n+2)/2;
  ll ans =0;
  for(int i=(n*k)-cnt;i>=0;i-=cnt){
    ans += a[i];
    k--;
    if(!k) break;
  }
  cout<<ans<<endl;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
