#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll n;
  cin>>n;
  vector<ll>a(n);
  for(ll i=0;i<n;i++)cin>>a[i];

  for(ll i=1;i<n-1;i++){
    if(a[i-1]<a[i] && a[i]>a[i+1]){
      cout<<"yes"<<endl;
      cout<<i<<" "<<i+1<<" "<<i+2<<endl;
      return;
    }
  } 
  cout<<"NO"<<endl;
  return; 
}

int main() {
