#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  string s;
  cin>>s;
  int n = s.size();
  int a=0,b=0;
  for(int i=0;i<n;i++){
    if(s[i]=='0')a++;
    else b++;
  }
  int mini = min(a,b);
  if(mini%2==1){
    cout<<"DA"<<endl;
    return;
  }
  else{
    cout<<"NET"<<endl;
    return;
  }
}

int main() {
#ifndef ONLINE_JUDGE

    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
