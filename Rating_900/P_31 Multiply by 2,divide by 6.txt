#include <bits/stdc++.h>
using namespace std;
#define ll long long 

void fun(){
  ll n;
  cin>>n;
  ll a =0,b=0;
  while(n>=1){
    if(n%2==0){
        n/=2;
        a++;
    }
    else if(n%3==0){
        n/=3;
        b++;
    }
    else if(n==1){
       break;
    }
    else{
        cout<<-1<<endl;
        return;
    }
  }
  if(b>=a){
    cout<<b-a+b<<endl;
    return;
  }
  else{
    cout<<-1<<endl;
  }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--) fun();
    return 0;
}
